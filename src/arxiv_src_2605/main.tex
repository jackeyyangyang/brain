% --- Delta-gain annotation (colored, smaller, lowered subscript at the bottom-right) ---
% \dpos / \dneg render a normal-width colored subscript and are used in tables
% where the subscript can extend freely to the right without being clipped
% (e.g. last-column transfer tables 4-6, ablations).
%
% \dvalp{NUM}{DELTA} / \dvaln{NUM}{DELTA} wrap a number AND its delta so the
% black NUMBER stays column-centered: an invisible left phantom of the same
% width as the right subscript is added, making the cell content symmetric
% around the number. This is used in the main results table where rows have
% a colored background (\cellcolor) and the subscript MUST live inside its
% own cell to avoid being painted over by the next cell's background.
\definecolor{deltapos}{HTML}{0E8A4A}
\definecolor{deltaneg}{HTML}{B91C1C}
